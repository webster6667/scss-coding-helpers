    fade-in

    @mixin fade-in($fade-in-duration: .5, $additional-transition-values: '') { ... }

    Description

    Show element from fade-out, and add transition style for fade-in after fade-out

    Parameters

    parameter Nameparameter Descriptionparameter Typeparameter Default value
    $fade-in-duration

    fade in duration

    		Number.5
    $additional-transition-values

    additional transition values

    		String''

    Example

    //fade-in
.player-fade-in {
  position: relative;
  height: 200px;
  background:silver;

  &__button {
    width: 10px;
    height: 10px;

    border-radius: 50%;
    background-color: black;

    @include fade-out(.5);
  }

  &:hover &__button {
    @include fade-in(.5);
  }

}

//fade-out
.player-fade-out {
  position: relative;
  height: 200px;
  background:silver;

  &__button {
    width: 10px;
    height: 10px;

    border-radius: 50%;
    background-color: black;

    @include fade-in(.5);
  }

  &:hover &__button {
    @include fade-out(.5);
  }

}

//fade effect with transform
.player-fade-in-rotate {
  position: relative;
  height: 200px;
  background:silver;

  $translateTransform: transform .5s 0s ease;

  &__button {
    width: 10px;
    height: 10px;

    background-color: black;

    transform: rotate(45deg);
    @include fade-out(.5, $translateTransform);
  }

  &:hover &__button {
    transform: rotate(0deg);
    @include fade-in(.5, $translateTransform);
  }

}

<div class="player-fade-in" >
  <div class="player-fade-in__button" ></div>
</div>

    fade-out

    @mixin fade-out($animation-duration: .5, $additional-transition-values: '') { ... }

    Description

    Hide element to opacity: 0, and hide element before fade-in

    Parameters

    parameter Nameparameter Descriptionparameter Typeparameter Default value
    $animation-duration

    duration of animation

    		Number.5
    $additional-transition-values

    additional transition values

    		String''

    Example

    .modal-trigger {
  height: 20px;
  background-color: silver;
}

//fade-in
.modal {
  position: fixed;
  top: 40px;
  right: 0;
  bottom: 0;
  left: 0;

  background-color: rgba(0,0,0,.5);
  @include fade-out(.5);

  &_open {
    @include fade-in(.5);
  }

}

//fade-out
.modal {
  position: absolute;
  top: 40px;
  right: 0;
  bottom: 0;
  left: 0;

  background-color: rgba(0,0,0,.5);
  @include fade-in(.5);

  &_closed {
    @include fade-out(.5);
  }

}

//fade effect with transform
.modal-rotate {
  $translateTransform: transform .5s 0s ease;

  position: absolute;
  top: 40px;
  right: 0;
  bottom: 0;
  left: 0;

  background-color: rgba(0,0,0,.5);
  transform: skewX(100deg);
  @include fade-out(.5, $translateTransform);

  &_open {
    transform: skewX(0deg);
    @include fade-in(.5, $translateTransform);
  }

}

    slide-in-down

    @mixin slide-in-down($animation-duration: .5, $has-fade-effect: false, $additional-transform-values: '') { ... }

    Description

    Slide element down

    Parameters

    parameter Nameparameter Descriptionparameter Typeparameter Default value
    $animation-duration

    slide-in and fade-in duration

    		Number.5
    $has-fade-effect

    should add fade effect

    		Booleanfalse
    $additional-transform-values

    additional transition values

    		String''

    Example

    .dd {
  margin: 40px auto;
  width: 300px;
  height: 40px * 3;
  overflow: hidden;

  $animation-in-duration: .5;
  $animation-out-duration: .5;
  $additional-transform-values: '';
  $has-fade-effect: true;
  $has-fade-out-delay: true;

  &__wrapper {
    position: relative;
  }

  &__active-item-wrapper {
    position: relative;
  }

  &_open &__list {
    @include slide-in-down()
  }

  &__list-item {
    height: 40px;
    background-color: silver;
    outline: 1px solid beige;
  }

  &__list {
    @include slide-out-down() {
      bottom: 0;
    }
  }

  &_with-fade & {

    &__list {
      @include slide-out-down($animation-out-duration, $has-fade-effect) {
        bottom: 0;
      }
    }

  }

  &_with-fade-out-delay & {

    &__list {
      @include slide-out-down($animation-out-duration, $has-fade-effect, $has-fade-out-delay) {
        bottom: 0;
      }
    }

  }

  &_with-fade#{&}_open &, &_with-fade-out-delay#{&}_open & {

    &__list {
      @include slide-in-down($animation-out-duration, $has-fade-effect)
    }

  }

}

    slide-out-down

    @mixin slide-out-down($animation-duration: .5, $has-fade-effect: false, $has-fade-out-delay: false, $fade-out-delay: $animation-duration, $additional-transform-values: '') { ... }

    Description

    Slide element up, from slide-in-down

    Parameters

    parameter Nameparameter Descriptionparameter Typeparameter Default value
    $animation-duration

    slide-in and fade-in duration

    		Number.5
    $has-fade-effect

    should add fade effect

    		Booleanfalse
    $has-fade-out-delay

    if true, element will hide, only after slide-down-out

    		Booleanfalse
    $fade-out-delay

    props for update fade-out delay after slide-down-out

    		Number$animation-duration
    $additional-transform-values

    additional transition values

    		String''

    Example

    .dd {
  margin: 40px auto;
  width: 300px;
  height: 40px * 3;
  overflow: hidden;

  &__wrapper {
    position: relative;
  }

  &__active-item-wrapper {
    position: relative;
  }

  &_open &__list {
    @include slideInDown()
  }

  &__list-item {
    height: 40px;
    background-color: silver;
    outline: 1px solid beige;
  }

  &__list {
    @include slideOutDown() {
      bottom: 0;
    }
  }

}

    absolute-center

    @mixin absolute-center($axis: both, $first-size: 0, $height: 0) { ... }

    Description

    Mixin that aligns inside a relative block, using absolute coordinates and negative margins

    Parameters

    parameter Nameparameter Descriptionparameter Typeparameter Default value
    $axis

    centering axis

    		Stringboth
    $first-size

    with or height

    		Number0
    $height

    block height

    		Number0

    Example

    $block-width: 50px;
$block-height: 50px;

.relative-block {
  position: relative;
  height: 400px;
  background-color: silver;
}

.center-block {
  width: $block-width;
  height: $block-height;

  font-size: 10px;
  line-height: $block-height;
  text-align: center;
  color: white;

  background-color: black;
  border-radius: 50%;

  //centring for x and y inside relative block
  &_relative-center-both {
    @include absolute-center(both, $block-width, $block-height);
  }

  //centring for y inside relative block
  &_relative-center-y {
    @include absolute-center(y, $block-height);
  }

  //centring for x inside relative block
  &_relative-center-x {
    @include absolute-center(x, $block-width);
  }

}

    transform-center

    @mixin transform-center($axis: both, $additionalTransform: '') { ... }

    Description

    Mixin that aligns inside a relative block, using transform: translate methods

    Parameters

    parameter Nameparameter Descriptionparameter Typeparameter Default value
    $axis

    centering axis

    		Stringboth
    $additionalTransform

    transform values that will be add to center translate values

    		String''

    Example

    .my-class {

  //centring for x and y inside relative block
  &_transform-center-both {
    @include transform-center();
  }

  //centring for y inside relative block
  &_transform-center-y {
    @include transform-center(y);
  }

  //centring for x inside relative block
  &_transform-center-x {
    @include transform-center(x);
  }

  //centring for x and y inside relative block, and rotate to 45deg
  &_transform-center-both-rotate-45 {
    @include transform-center(both, rotate(45deg));
  }

}
